export function existsSync(path) {
    return puer.fileExists(path);
}

export function readFileSync(path) {
    return puer.loadFile(path).content;
}

export function readFile(name, callback) {
    const content = puer.loadFile(name).content;
    if (content) {
        callback?.(null, content);
    } else {
        callback?.(new Error(`file not exists: ${name}`), null);
    }

}

module.exports = exports = global.fs ??= { existsSync, readFileSync, readFile }
